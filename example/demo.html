        throw new Error("Eventually handled");
    })

    Q.delay(2000)
    .thenResolve(rejection)
    .catch(function () {
        // handled
    });

    var deferred = Q.defer(); // ...suspense
    setInterval(function () {
        deferred.notify(Math.random());
    }, 1000);

    var promises = [];
    for (var i = 0; i < 10; i++) {
        promises.push(Q.delay(10000 * Math.random()));
    }
    Q.all(promises);
